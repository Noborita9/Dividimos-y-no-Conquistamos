\rlabel{Necklaces under cyclic $C_n$ (gcd form):}
\[
N_{\text{rot}}=\frac{1}{n}\sum_{k=0}^{n-1} m^{\gcd(n,k)}.
\]

\rlabel{Equivalent divisor form (for $C_n$):}
\[
N_{\text{rot}}=\frac{1}{n}\sum_{d\mid n}\varphi(d)\,m^{\,n/d}.
\]

\rlabel{Necklaces/bracelets under dihedral $D_n$ (gcd form):}
\[
N =
\begin{cases}
\displaystyle \frac{1}{2n}\!\left(\sum_{k=0}^{n-1} m^{\gcd(n,k)} + n\,m^{(n+1)/2}\right), & n \text{ odd},\\[1.2ex]
\displaystyle \frac{1}{2n}\!\left(\sum_{k=0}^{n-1} m^{\gcd(n,k)} + \tfrac{n}{2}\,m^{n/2} + \tfrac{n}{2}\,m^{n/2+1}\right), & n \text{ even}.
\end{cases}
\]

\rtext{Here $\gcd$ is the greatest common divisor. A rotation by $k$ positions has $\gcd(n,k)$ cycles.}
